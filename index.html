    <script>

        function calculateNumber() {

            const birthDay = document.getElementById('birthDay').value;

            if (!birthDay || birthDay < 1 || birthDay > 31) {

                document.getElementById('result').innerText = "Введите корректный день (1-31).";

                return;

            }


            let sum = parseInt(birthDay);


            // Сводим к одной цифре

            while (sum > 9) {

                sum = sum.toString().split('').reduce((acc, digit) => acc + parseInt(digit), 0);

            }


            document.getElementById('result').innerText = `Ваше число: ${sum}`;

        }

    </script>
